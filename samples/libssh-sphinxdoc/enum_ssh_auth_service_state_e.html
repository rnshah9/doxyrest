#include <auth.h>

enum ssh_auth_service_state_e {
    SSH_AUTH_SERVICE_NONE      =0,
    SSH_AUTH_SERVICE_SENT,
    SSH_AUTH_SERVICE_ACCEPTED,
    SSH_AUTH_SERVICE_DENIED,
    SSH_AUTH_SERVICE_USER_SENT,
};
